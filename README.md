allprojects {

}

}

dependencies {

....

implementation 'com.github.bigfunglobal:TopGameSDK:v1.0.2'

}

TopGameSDK.init(this,adjustAppToken,talkingDataAppId,talkingDataChannelCode,);//初始化的接口重写
